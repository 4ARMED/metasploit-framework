def initialize(framework, opts)
  super
  self.framework.events.add_exploit_subscriber(self)
  self.framework.events.add_session_subscriber(self)
  self.framework.events.add_general_subscriber(self)
  self.framework.events.add_db_subscriber(self)
  self.framework.events.add_ui_subscriber(self)
  self.framework.datastore['CCDCXXX_SERVER']    = "https://ccdc.xxx/msf"
  self.framework.datastore['CCDCXXX_USER']      = "egypt"
  self.framework.datastore['CCDCXXX_EVENT']     = "nccdc"
  self.framework.datastore['CCDCXXX_API_TOKEN'] = "enter_api_token_here"
  self.framework.datastore['CCDCXXX_METRIC']    = true
  self.framework.datastore['CCDCXXX_DEBUG']     = false
  add_console_dispatcher(CCDCXXXCommandDispatcher)
  print_status("CCDCXXX plugin loaded. Please configure your variables!")
end

def ccdc_post(endpoint, postdata)
  begin
    uri              = URI.parse(self.framework.datastore['CCDCXXX_SERVER'] + endpoint)
    http             = Net::HTTP.new(uri.host, uri.port)
    http.use_ssl     = true
    http.verify_mode = OpenSSL::SSL::VERIFY_NONE
    req              = Net::HTTP::Post.new(uri.path)

    req.set_form_data(postdata)
    resp = http.request(req)
    return_status = JSON.parse(resp.body)
    if return_status.key? 'error'
      print_error("CCDC.XXX API Error: #{return_status['error']}")
    end
    return resp
  rescue Exception => e
    print_line "ERROR! #{e}"
  end
end

def on_exploit_success(mod, session)
  begin
    http = self.ccdc_post('/exploit_success', {
      'event_id'           => self.framework.datastore['CCDCXXX_EVENT'],
      'user'               => self.framework.datastore['CCDCXXX_USER'],
      'token'              => self.framework.datastore['CCDCXXX_API_TOKEN'],
      'player_ip'          => mod.datastore['LHOST'],
      'module'             => mod.fullname,
      'target_ip'          => mod.datastore['RHOST'],
      'details'            => session.type
    })
    if (self.framework.datastore['CCDCXXX_DEBUG'] == "true")
      self.status(http.body)
    end
  rescue
    self.status("error")
  end
  return
end

def on_session_command(session, cmd)
  target,vport = session.tunnel_peer.split(':')
  if cmd == "exit"
    begin
      http = self.ccdc_post('/session_exit', {
        'event_id'       => self.framework.datastore['CCDCXXX_EVENT'],
        'user'           => self.framework.datastore['CCDCXXX_USER'],
        'token'          => self.framework.datastore['CCDCXXX_API_TOKEN'],
        'player_ip'      => session.session_host,
        'module'         => session.type,
        'target_ip'      => target
      })
    if (self.framework.datastore['CCDCXXX_DEBUG'] == "true")
      self.status(http.body)
    end
    rescue
      self.status("error")
    end
  elsif (self.framework.datastore['CCDCXXX_METRIC'] == "true")
    begin
      http = self.ccdc_post('/session_command', {
        'event_id'       => self.framework.datastore['CCDCXXX_EVENT'],
        'user'           => self.framework.datastore['CCDCXXX_USER'],
        'token'          => self.framework.datastore['CCDCXXX_API_TOKEN'],
        'player_ip'      => session.session_host,
        'module'         => session.type,
        'target_ip'      => target,
        'details'        => cmd
      })
    if (self.framework.datastore['CCDCXXX_DEBUG'] == "true")
      self.status(http.body)
    end
    rescue Exception => e
      self.status("error")
    end
  end
  return
end
