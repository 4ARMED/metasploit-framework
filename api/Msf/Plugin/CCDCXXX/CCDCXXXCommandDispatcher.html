def ccdc_post(endpoint, postdata)
  begin
    uri              = URI.parse(self.framework.datastore['CCDCXXX_SERVER'] + endpoint)
    http             = Net::HTTP.new(uri.host, uri.port)
    http.use_ssl     = true
    http.verify_mode = OpenSSL::SSL::VERIFY_NONE
    req              = Net::HTTP::Post.new(uri.path)

    req.set_form_data(postdata)
    resp = http.request(req)
    return_status = JSON.parse(resp.body)
    if return_status.key? 'error'
      print_error("CCDC.XXX API Error: #{return_status['error']}")
    end
    return resp
  rescue Exception => e
    print_line "ERROR! #{e}"
  end
end

def cmd_ccdcxxx_test_auth(*args)
  begin
    http = self.ccdc_post('/test_auth', {
      'event_id'          => self.framework.datastore['CCDCXXX_EVENT'],
      'user'              => self.framework.datastore['CCDCXXX_USER'],
      'token'             => self.framework.datastore['CCDCXXX_API_TOKEN']
    })
    if (self.framework.datastore['CCDCXXX_DEBUG'] == "true")
      self.status(http.body)
    end
    print_status("Successfully authenticated with the CCDC.XX API.")
  rescue Exception => e
    print_error("Exception: #{e}")
    self.status("error")
  end
end
