def negotiate_connection
  @version = @sock.read(1).unpack("C").first
  case @version
  when 4
    negotiate_connection_4
  when 5
    negotiate_connection_5
  else
    raise UnknownProtocolVersionError
  end
end

def negotiate_connection_4
  header = ''
  # The whole header is actually 8 bytes, but we already read the
  # version byte.
  while header.length < 7
    header << @lsock.read(1)
  end
  @user_id = read_until_null
  header << @user_id

end
